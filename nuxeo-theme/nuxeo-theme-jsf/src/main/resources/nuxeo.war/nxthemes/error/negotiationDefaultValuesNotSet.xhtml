application configuration error

Default negotiation settings are not properly configured for this application, please set the default theme, the default engine and the default perspective.

Example: 

<extension target="org.nuxeo.theme.services.ThemeService" point="applications">

  <application root="/nuxeo">

     <negotiation>
      <default-theme>default/default</default-theme>
      <default-perspective>default</default-perspective>
      <default-engine>default</default-engine>
    </negotiation>

  </application>

</extension>
